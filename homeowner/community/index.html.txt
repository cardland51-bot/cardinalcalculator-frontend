    // Handle file label
    const photoInput = document.getElementById("photoInput");
    const fileLabel = document.getElementById("fileLabel");
    if (photoInput) {
      photoInput.addEventListener("change", () => {
        const file = photoInput.files[0];
        fileLabel.textContent = file ? "Photo added" : "Add photo (optional)";
      });
    }

    // Simple "post to board" demo (front-end only)
    const postForm = document.getElementById("postForm");
    const postText = document.getElementById("postText");
    const tagSelect = document.getElementById("tagSelect");
    const threadList = document.getElementById("threadList");

    if (postForm && postText && threadList) {
      postForm.addEventListener("submit", (e) => {
        e.preventDefault();
        const text = postText.value.trim();
        const tag = tagSelect.value;
        if (!text) return;

        const div = document.createElement("div");
        div.className = "thread";
        const label = tag
          ? `${tag} â€¢ New from the board`
          : "New yard question";
        div.innerHTML = `
          <div class="thread-label">${label}</div>
          <div class="thread-question">${text}</div>
          <div class="thread-detail">
            Waiting on replies from other readers. Check back soon.
          </div>
          <div class="thread-replies">0 replies yet</div>
        `;
        threadList.prepend(div);

        postText.value = "";
        tagSelect.value = "";
        if (photoInput) {
          photoInput.value = "";
          fileLabel.textContent = "Add photo (optional)";
        }
      });
    }

    // Poll interactions (front-end only)
    const pollOptions = document.getElementById("pollOptions");
    const pollSubmit = document.getElementById("pollSubmit");
    const pollNote = document.getElementById("pollNote");
    const pollFreeform = document.getElementById("pollFreeform");

    if (pollOptions && pollSubmit && pollNote) {
      pollOptions.addEventListener("click", (e) => {
        const opt = e.target.closest(".poll-option");
        if (!opt) return;
        document
          .querySelectorAll(".poll-option")
          .forEach((el) => el.classList.remove("active"));
        opt.classList.add("active");
      });

      pollSubmit.addEventListener("click", () => {
        const active = document.querySelector(".poll-option.active");
        const choice = active ? active.dataset.value : null;
        const idea = (pollFreeform && pollFreeform.value.trim()) || "";
        // demo: just acknowledge
        pollNote.textContent =
          "Thanks. Your pick just went into the next Digest brainstorm.";
        if (pollFreeform) pollFreeform.value = "";
        if (active) active.classList.remove("active");
        setTimeout(() => {
          pollNote.textContent =
            "Ideas stay anonymous. We use them to shape future articles and tools.";
        }, 3500);
      });
    }
