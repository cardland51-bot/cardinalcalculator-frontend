  <script>
    // File label
    const fileInput = document.getElementById("fileInput");
    const fileText = document.getElementById("fileText");
    if (fileInput && fileText) {
      fileInput.addEventListener("change", () => {
        const file = fileInput.files[0];
        fileText.textContent = file ? "Pic ready" : "Add pic";
      });
    }

    // Template prompts
    const promptPills = document.querySelectorAll(".prompt-pill");
    const captionInput = document.getElementById("captionInput");
    promptPills.forEach(pill => {
      pill.addEventListener("click", () => {
        const fill = pill.getAttribute("data-fill");
        if (fill && captionInput) captionInput.value = fill;
      });
    });

    // Post meme (front-end only demo)
    const postBtn = document.getElementById("postMemeBtn");
    const memeList = document.getElementById("memeList");
    const templateSelect = document.getElementById("templateSelect");

    if (postBtn && memeList && captionInput) {
      postBtn.addEventListener("click", () => {
        const caption = captionInput.value.trim();
        if (!caption) return;

        const tpl = templateSelect.value;
        let tag = "other";
        if (tpl === "quick-mow") tag = "client";
        if (tpl === "before-after") tag = "stripe";
        if (tpl === "schedule") tag = "client";
        if (tpl === "weather") tag = "weather";

        const card = document.createElement("div");
        card.className = "meme-card";
        card.dataset.tag = tag;

        const initials = "PRO";

        card.innerHTML = `
          <div class="meme-avatar">${initials}</div>
          <div>
            <div class="meme-tag">${tag === "other" ? "From the route" : tag}</div>
            <div class="meme-caption">${caption}</div>
            <div class="meme-body">
              Posted via Moe Mode demo composer.
            </div>
            <div class="meme-emotes">
              <div class="meme-react" data-type="lol">
                <span class="icon">üòÇ</span><span class="count">0</span>
              </div>
              <div class="meme-react" data-type="relate">
                <span class="icon">ü§ù</span><span class="count">0</span>
              </div>
              <div class="meme-label">Just now</div>
            </div>
          </div>
        `;
        memeList.prepend(card);

        captionInput.value = "";
        templateSelect.value = "";
        if (fileInput) {
          fileInput.value = "";
          fileText.textContent = "Add pic";
        }
      });
    }

    // Reaction bumps (front-end only)
    document.addEventListener("click", (e) => {
      const react = e.target.closest(".meme-react");
      if (!react) return;
      const countEl = react.querySelector(".count");
      if (!countEl) return;
      const current = parseInt(countEl.textContent || "0", 10);
      countEl.textContent = current + 1;
    });

    // Tag filter
    const chips = document.querySelectorAll(".chip");
    chips.forEach(chip => {
      chip.addEventListener("click", () => {
        const tag = chip.dataset.tag;
        chips.forEach(c => c.classList.remove("active"));
        chip.classList.add("active");

        document.querySelectorAll(".meme-card").forEach(card => {
          if (!tag || tag === "all") {
            card.style.display = "";
          } else {
            card.style.display =
              card.dataset.tag === tag ? "" : "none";
          }
        });
      });
    });
  </script>
