    nightToggle.addEventListener("click", () => {

      nightToggle.classList.toggle("active", on);
    });

    // Fullscreen toggle
    fullscreenToggle.addEventListener("click", () => {
      const on = shell.classList.toggle("fullscreen");
      fullscreenToggle.classList.toggle("active", on);
    });

    // Language toggle (simple demo: just toggles label & one line)
    let lang = "en";
    langToggle.addEventListener("click", () => {
      lang = lang === "en" ? "es" : "en";
      langToggle.classList.toggle("active", lang === "es");
      if (lang === "es") {
        langToggle.textContent = "ES / EN";
      } else {
        langToggle.textContent = "EN / ES";
      }

    });

      aiBubble.classList.add("hidden");
    });

    setTimeout(() => {
      aiBubble.classList.add("hidden");
    }, 12000);

    function hidePaywall() {
      paywallOverlay.classList.add("hidden");
    }

    paywallUnlock.addEventListener("click", hidePaywall);
    unlockBtn.addEventListener("click", hidePaywall);

    // Book click -> load lesson
    bookGrid.addEventListener("click", (e) => {
      const book = e.target.closest(".book");
      if (!book) return;

      document.querySelectorAll(".book").forEach(b => b.classList.remove("active"));
      book.classList.add("active");

      const title = book.getAttribute("data-title") || "";
      const body = book.getAttribute("data-body") || "";

      lessonTitle.textContent = title;
      lessonBody.textContent = body;
