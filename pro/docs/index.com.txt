<script>
  // THEME TOGGLE
  (function(){
    const saved = localStorage.getItem("cgDocsTheme");
    if (saved && saved !== "default") {
      document.body.classList.add("theme-" + saved);
      document.querySelectorAll("#themePills .pill-btn").forEach(b=>{
        b.classList.toggle("active", b.dataset.theme === saved);
      });
    }
    document.querySelectorAll("#themePills .pill-btn").forEach(btn=>{
      btn.addEventListener("click",()=>{
        document.querySelectorAll("#themePills .pill-btn").forEach(b=>b.classList.remove("active"));
        btn.classList.add("active");
        document.body.classList.remove("theme-earth","theme-night","theme-green");
        const t = btn.dataset.theme;
        if (t && t !== "default") document.body.classList.add("theme-" + t);
        localStorage.setItem("cgDocsTheme", t || "default");
      });
    });
  })();

  // TABS
  (function(){
    const tabs = document.querySelectorAll(".tab-btn");
    const panels = {
      estimates: document.getElementById("tab-estimates"),
      site: document.getElementById("tab-site"),
      client: document.getElementById("tab-client"),
      ops: document.getElementById("tab-ops"),
      salesu: document.getElementById("tab-salesu")
    };
    tabs.forEach(btn=>{
      btn.addEventListener("click",()=>{
        tabs.forEach(b=>b.classList.remove("active"));
        btn.classList.add("active");
        const key = btn.dataset.tab;
        Object.keys(panels).forEach(k=>{
          panels[k].style.display = (k === key) ? "grid" : "none";
        });
      });
    });
  })();

  // "Add to home" helper (we can't force install, just educate)
  (function(){
    const btn = document.getElementById("btnShortcut");
    const toast = document.getElementById("shortcutToast");
    const close = document.getElementById("toastClose");
    if (!btn || !toast || !close) return;

    btn.addEventListener("click", ()=>{
      toast.style.display = "flex";
    });
    close.addEventListener("click", ()=>{
      toast.style.display = "none";
    });
  })();
</script>
